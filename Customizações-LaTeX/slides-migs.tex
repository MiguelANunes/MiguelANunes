% Slide que mostra a nova seção atual quando ela mudo para ela, de modo mais controlado
\def\SecT{
    \begin{frame}
        \centering{\huge\thesection. \insertsection}

        \vspace{\baselineskip}
    \end{frame}
}

% Slide que mostra a nova subseção atual quando ela mudo para ela, de modo mais controlado
\def\SubSecT{
    \begin{frame}
        \centering{\huge\thesection.\thesubsection\ \insertsubsection}

        \vspace{\baselineskip}
    \end{frame}
}

% Suprimindo warnings do hyperref
\pdfstringdefDisableCommands{%
  \def\\{}%
  \def\texttt#1{#1}%
  \def\smallskip{}%
  \def\medskip{}%
}
