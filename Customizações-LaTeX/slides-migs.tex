% Começando seções no 0
\makeatletter
\patchcmd{\beamer@sectionintoc}
  {\ifnum\beamer@tempcount>0}
  {\ifnum\beamer@tempcount>-1}
  {}
  {}
\beamer@tocsectionnumber=-1
\makeatother
\setcounter{section}{-1}

% Slide que mostra a nova seção atual quando ela mudo para ela, de modo mais controlado
\def\SecT{
    \begin{frame}
        \centering{\huge\thesection. \insertsection}

        \vspace{\baselineskip}
    \end{frame}
}

\def\SubSecT{
    \begin{frame}
        \centering{\huge\thesection.\thesubsection\ \insertsubsection}

        \vspace{\baselineskip}
    \end{frame}
}

% Suprimindo warnings do hyperref
\pdfstringdefDisableCommands{%
  \def\\{}%
  \def\texttt#1{#1}%
  \def\smallskip{}%
  \def\medskip{}%
}

% Texto que será apresentado apenas no próximo slide fica transparente ao invés de invisível
\setbeamercovered{transparent}
